pipeline {
    agent {
        label 'android'
    }
    options {
        skipStagesAfterUnstable()
    }
    stages {
        stage('CheckOut') {
            steps {
                checkout([$class: 'GitSCM', branches: [[name: '*/master']], extensions: [], userRemoteConfigs: [[url: 'https://github.com/TanishqSehgal7/QuizzApp']]])
            }
        }
        stage('Compile') {
            steps {
             sh './gradlew compileDebugSources'
            }
        }
        stage('Unit Test') {
            steps {
                sh './gradlew testDebugUnitTest'

            }
        }
         stage('Build APK') {
      steps {
        // Finish building and packaging the APK
        sh './gradlew assembleDebug'

        // Archive the APKs so that they can be downloaded from Jenkins
        archiveArtifacts '**/*.apk'
          }
       }
    }
}
